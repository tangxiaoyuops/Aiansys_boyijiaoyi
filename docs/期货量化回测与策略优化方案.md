**2. 风险指标计算模块**：

```python
# core/backtest/metrics/risk.py
import numpy as np
import pandas as pd
from typing import List, Tuple

class RiskMetrics:
    """风险指标计算"""
    
    def __init__(self, equity_curve: List[float]):
        self.equity_curve = np.array(equity_curve)
        self.returns = np.diff(self.equity_curve) / self.equity_curve[:-1]
    
    def max_drawdown(self) -> Dict[str, float]:
        """
        计算最大回撤及相关指标
        
        Returns:
            {
                'max_drawdown': 最大回撤百分比,
                'max_drawdown_duration': 最大回撤持续天数,
                'current_drawdown': 当前回撤,
                'drawdown_series': 回撤序列
            }
        """
        cumulative = np.maximum.accumulate(self.equity_curve)
        drawdown = (self.equity_curve - cumulative) / cumulative
        
        max_dd = np.min(drawdown)
        max_dd_idx = np.argmin(drawdown)
        
        # 计算回撤持续时间
        peak_idx = np.argmax(self.equity_curve[:max_dd_idx+1])
        dd_duration = max_dd_idx - peak_idx
        
        # 当前回撤
        current_dd = drawdown[-1]
        
        return {
            'max_drawdown': float(max_dd * 100),  # 百分比
            'max_drawdown_duration': int(dd_duration),
            'current_drawdown': float(current_dd * 100),
            'drawdown_series': drawdown.tolist()
        }
    
    def volatility(self, annualize: bool = True) -> float:
        """
        计算波动率
        
        Args:
            annualize: 是否年化
            
        Returns:
            波动率（百分比）
        """
        vol = np.std(self.returns)
        if annualize:
            # 假设252个交易日
            vol = vol * np.sqrt(252)
        return float(vol * 100)
    
    def downside_deviation(self, target_return: float = 0.0) -> float:
        """
        计算下行偏差（用于Sortino比率）
        
        Args:
            target_return: 目标收益率
            
        Returns:
            下行偏差
        """
        downside_returns = self.returns[self.returns < target_return]
        if len(downside_returns) == 0:
            return 0.0
        return float(np.std(downside_returns) * np.sqrt(252) * 100)
    
    def value_at_risk(self, confidence: float = 0.95) -> float:
        """
        计算VaR（Value at Risk）
        
        Args:
            confidence: 置信水平（如0.95表示95%置信度）
            
        Returns:
            VaR值（百分比）
        """
        return float(np.percentile(self.returns, (1 - confidence) * 100) * 100)
    
    def conditional_var(self, confidence: float = 0.95) -> float:
        """
        计算CVaR（条件风险价值，期望损失）
        
        Args:
            confidence: 置信水平
            
        Returns:
            CVaR值（百分比）
        """
        var = np.percentile(self.returns, (1 - confidence) * 100)
        cvar = self.returns[self.returns <= var].mean()
        return float(cvar * 100)
    
    def ulcer_index(self) -> float:
        """
        计算溃疡指数（Ulcer Index）
        衡量回撤的深度和持续时间
        
        Returns:
            溃疡指数
        """
        cumulative = np.maximum.accumulate(self.equity_curve)
        drawdown = (self.equity_curve - cumulative) / cumulative
        return float(np.sqrt(np.mean(drawdown ** 2)) * 100)
```

**凯利公式应用**：
```python
# 凯利公式：f* = (bp - q) / b
# f*: 最优投注比例
# b: 盈亏比
# p: 胜率
# q: 败率 = 1 - p

def kelly_criterion(win_rate: float, profit_loss_ratio: float) -> float:
    """
    计算凯利比例
    
    Args:
        win_rate: 胜率
        profit_loss_ratio: 盈亏比
        
    Returns:
        最优仓位比例
    """
    b = profit_loss_ratio
    p = win_rate
    q = 1 - p
    
    kelly = (b * p - q) / b
    
    # 半凯利（保守）
    half_kelly = kelly * 0.5
    
    # 限制最大比例
    return max(0, min(half_kelly, 0.25))
```

```python
# core/strategy/base.py
from abc import ABC, abstractmethod
from typing import Dict, Any, Optional
import pandas as pd

class Strategy(ABC):
    """策略基类"""
    
    def __init__(self, name: str = "Base Strategy"):
        self.name = name
        self.parameters = {}
        self.initialized = False
        
    def initialize(self, parameters: Dict[str, Any]):
        """
        初始化策略参数
        
        Args:
            parameters: 策略参数字典
        """
        self.parameters = parameters
        self.initialized = True
        self._on_initialize()
    
    @abstractmethod
    def _on_initialize(self):
        """子类实现：初始化逻辑"""
        pass
    
    @abstractmethod
    def generate_signal(
        self,
        data: pd.DataFrame,
        portfolio: 'Portfolio'
    ) -> Optional[Dict[str, Any]]:
        """
        生成交易信号
        
        Args:
            data: 历史数据
            portfolio: 当前组合状态
            
        Returns:
            信号字典 {
                'action': 'OPEN_LONG' | 'OPEN_SHORT' | 'CLOSE_LONG' | 'CLOSE_SHORT',
                'size': 交易数量,
                'reason': 交易原因
            }
            或 None（无信号）
        """
        pass
    
    def get_parameter(self, key: str, default: Any = None) -> Any:
        """获取参数值"""
        return self.parameters.get(key, default)

# 具体策略示例
class DualMAStrategy(Strategy):
    """双均线策略"""
    
    def _on_initialize(self):
        self.fast_period = self.get_parameter('fast_period', 5)
        self.slow_period = self.get_parameter('slow_period', 20)
        self.position_size = self.get_parameter('position_size', 0.3)
        
    def generate_signal(self, data: pd.DataFrame, portfolio) -> Optional[Dict]:
        if len(data) < self.slow_period:
            return None
        
        # 计算均线
        fast_ma = data['close'].rolling(self.fast_period).mean().iloc[-1]
        slow_ma = data['close'].rolling(self.slow_period).mean().iloc[-1]
        prev_fast = data['close'].rolling(self.fast_period).mean().iloc[-2]
        prev_slow = data['close'].rolling(self.slow_period).mean().iloc[-2]
        
        # 金叉
        if prev_fast <= prev_slow and fast_ma > slow_ma:
            if portfolio.position <= 0:  # 空仓或持空
                return {
                    'action': 'OPEN_LONG',
                    'size': self.position_size,
                    'reason': f'金叉：MA{self.fast_period}突破MA{self.slow_period}'
                }
        
        # 死叉
        elif prev_fast >= prev_slow and fast_ma < slow_ma:
            if portfolio.position >= 0:  # 空仓或持多
                return {
                    'action': 'OPEN_SHORT',
                    'size': self.position_size,
                    'reason': f'死叉：MA{self.fast_period}跌破MA{self.slow_period}'
                }
        
        return None
```

```python
import multiprocessing

def parallel_backtest(
    data: pd.DataFrame,
    strategy: Strategy,
    param_list: List[Dict]
) -> List[Dict]:
    """并行运行多个参数配置的回测"""
    
    num_workers = multiprocessing.cpu_count() - 1
    
    with ProcessPoolExecutor(max_workers=num_workers) as executor:
        futures = [
```

**3. 增量回测**：
```python
class IncrementalBacktest:
    """增量回测：只回测新增数据"""
    
    def __init__(self, engine: FuturesBacktestEngine):
        self.engine = engine
        self.last_index = 0
        self.state = None
    
    def update(self, new_data: pd.DataFrame):
        """更新新数据"""
        if self.state is None:
            # 首次回测
            result = self.engine.run_backtest(new_data, strategy)
            self.state = result['state']
            self.last_index = len(new_data)
        else:
            # 增量回测
            incremental_data = new_data.iloc[self.last_index:]
            result = self.engine.run_incremental(
                incremental_data,
                strategy,
                initial_state=self.state
            )
            self.state = result['state']
            self.last_index = len(new_data)
        
        return result
```

### B. 常见问题

**Q1: 回测结果与实盘差异大？**
A: 考虑滑点、手续费、市场冲击、流动性等因素，使用保守参数。

**Q2: 如何避免过拟合？**
A: 使用样本外测试、交叉验证、参数敏感性分析。

**Q3: 策略优化需要多长时间？**
A: 取决于参数空间大小和优化算法，通常几分钟到几小时不等。
